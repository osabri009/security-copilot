openapi: 3.0.0

info:
  title: FortiAnalyzer Threats API
  description: Retrieve threat information from FortiAnalyzer
  version: "v1"

servers:
  - url: https://<your-fortianalyzer-url>/api/v2

paths:
  /threats:
    get:
      operationId: getThreats
      summary: Retrieve threat information from FortiAnalyzer
      parameters:
        - in: query
          name: threatId
          schema:
            type: string
          required: false
          description: A specific threat ID to retrieve details for a particular threat.
        - in: query
          name: severity
          schema:
            type: string
            enum: [low, medium, high, critical]
          required: false
          description: Filter threats based on severity level.
        - in: query
          name: startTime
          schema:
            type: string
            format: date-time
          required: false
          description: Start time for filtering threat events.
        - in: query
          name: endTime
          schema:
            type: string
            format: date-time
          required: false
          description: End time for filtering threat events.
      responses:
        "200":
          description: A list of threat events
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/ThreatResponse"

components:
  schemas:
    ThreatResponse:
      type: object
      properties:
        threats:
          type: array
          items:
            type: object
            properties:
              threatId:
                type: string
                description: Unique identifier of the threat.
              severity:
                type: string
                description: Severity of the threat.
              description:
                type: string
                description: Detailed description of the threat.
              timestamp:
                type: string
                format: date-time
                description: When the threat was detected.
              status:
                type: string
                description: Current status of the threat (e.g., active, mitigated).
